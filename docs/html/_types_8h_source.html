228 enum class SendFlag : uint8_t
229 {
230 // Reliable message send. Can send up to k_cbMaxSteamNetworkingSocketsMessageSizeSend bytes in a single message.
231 // Does fragmentation/re-assembly of messages under the hood, as well as a sliding window for
232 // efficient sends of large chunks of data.
233 //
234 // The Nagle algorithm is used.
235 RELIABLE = 8,
236 // Send the message unreliably. Can be lost. Messages *can* be larger than a
237 // single MTU (UDP packet), but there is no retransmission, so if any piece
238 // of the message is lost, the entire message will be dropped.
239 //
240 // The sending API does have some knowledge of the underlying connection, so
241 // if there is no NAT-traversal accomplished or there is a recognized adjustment
242 // happening on the connection, the packet will be batched until the connection
243 // is open again.
244 UN_RELIABLE = 0,
245 };
